function Lol() {
    let oracion = prompt("Ingrese la frase.");
    if(!oracion){
        alert("Error: Ingrese la frase.");
        return;
    }
    let conteo = 0;
    for(let i = 0; i < oracion.length; i++){
        conteo++
    }
    alert(`La frase tiene ${conteo} caracteres.`)

}   
Lol()
