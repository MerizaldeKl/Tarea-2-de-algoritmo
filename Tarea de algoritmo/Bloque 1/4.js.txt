function Contar() {
  let ciudad1 = prompt("Ingrese la primer ciudad:");
  let ciudad2 = prompt("Ingrese la segunda ciudad:");

  if (!ciudad1 || !ciudad2) {
    alert("No ingresaste una o ambas ciudades.");
    return;
  }

  if (ciudad1.length > ciudad2.length) {
    alert(`${ciudad1} tiene m치s letras.`);
  } else if (ciudad2.length > ciudad1.length) {
    alert(`${ciudad2} tiene m치s letras.`);
  } else {
    alert(`Las ciudades ${ciudad1} y ${ciudad2} tienen la misma cantidad de letras.`);
  }
}
Contar()
