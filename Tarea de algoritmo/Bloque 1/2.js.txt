function Ey() {
    let frase = prompt("Ingrese la frase");
    let letra = prompt("Ingrese la letra.");
    if(!frase | !letra){
        alert("Error ingrese la frase o letra.");
        return;
    }
    let aparicion = 0
    for(let i = 0; i < frase.length; i++){
        if(frase[i].toLowerCase() == letra.toLowerCase()){
            aparicion++
    } 
}
alert(`La letra ${letra} aparece ${aparicion} veces en la oracion.`)
}
Ey();
