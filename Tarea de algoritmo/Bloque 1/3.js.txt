function invertido() {
    let palabra = prompt("Ingrese la palabra.")
    if(!palabra){
        alert("Error: Ingrese la palabra.");
        return;
    }
    let invertido = ""
    for(let i = palabra.length - 1; i >= 0; i--){
        invertido += palabra[i];
    }
    alert(`El texto queda asi: ${invertido.toLowerCase()}.`)
}
invertido()
