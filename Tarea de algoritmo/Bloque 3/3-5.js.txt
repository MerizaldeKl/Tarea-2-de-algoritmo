function Promedios() {
  let n = parseInt(prompt("Ingrese la cantidad de números:"));
  
  if (isNaN(n) || n <= 0) {
    alert("Error: Ingrese una cantidad válida.");
    return;
  }

  let sumaPares = 0, sumaImpares = 0;
  let contPares = 0, contImpares = 0;

  for (let i = 1; i <= n; i++) {
    let num = parseInt(prompt(`Ingrese el número ${i}:`));

    if (num % 2 === 0) {
      sumaPares += num;
      contPares++;
    } else {
      sumaImpares += num;
      contImpares++;
    }
  }

  let promPares = contPares > 0 ? (sumaPares / contPares).toFixed(2) : 0;
  let promImpares = contImpares > 0 ? (sumaImpares / contImpares).toFixed(2) : 0;

  alert(`Promedio de pares: ${promPares}\nPromedio de impares: ${promImpares}`);
}

Promedios();
