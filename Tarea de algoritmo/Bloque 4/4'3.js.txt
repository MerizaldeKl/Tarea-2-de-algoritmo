//Inicio
//  edades ← [17, 20, 35, 15, 18]
//  sumaMayores ← 0
//  sumaMenores ← 0
//  contMayores ← 0
//  contMenores ← 0

//  Para i ← 1 Hasta longitud(edades) Hacer
//    Si edades[i] ≥ 18 Entonces
//      sumaMayores ← sumaMayores + edades[i]
//      contMayores ← contMayores + 1
//    Sino
//      sumaMenores ← sumaMenores + edades[i]
//      contMenores ← contMenores + 1
//    FinSi
//  FinPara

//  promedioMayores ← sumaMayores / contMayores
//  promedioMenores ← sumaMenores / contMenores

//  Escribir "Promedio de mayores de edad: ", promedioMayores
//  Escribir "Promedio de menores de edad: ", promedioMenores
//Fin
function promedioEdades() {
  let edades = [17, 20, 35, 15, 18];
  let sumaMayores = 0, sumaMenores = 0;
  let contMayores = 0, contMenores = 0;

  for (let i = 0; i < edades.length; i++) {
    if (edades[i] >= 18) {
      sumaMayores += edades[i];
      contMayores++;
    } else {
      sumaMenores += edades[i];
      contMenores++;
    }
  }

  let promedioMayores = contMayores > 0 ? (sumaMayores / contMayores) : 0;
  let promedioMenores = contMenores > 0 ? (sumaMenores / contMenores) : 0;

  console.log("Promedio de mayores de edad: " + promedioMayores.toFixed(2));
  console.log("Promedio de menores de edad: " + promedioMenores.toFixed(2));
}

promedioEdades();
